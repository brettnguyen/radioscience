"""Configuration loading and validation for the COSMIC-2 BP reproduction.

This module maps the nested YAML configuration used by the repository into the
flat ``AppConfig`` interface specified by the project design. The public
``AppConfig`` intentionally exposes only the fields currently required by the
processing modules.

The configuration values implemented here are paper-backed defaults from the
provided ``config.yaml`` and the described COSMIC-2 back-propagation method.
External data directories are converted to ``Path`` objects but are not required
to exist at load time, so synthetic-only runs can be configured without having
downloaded COSMIC-2 or GNSS products.
"""

class _MagneticFieldConfig(_StrictBaseModel):
    """Magnetic-field model configuration."""

    model: str = Field(default="IGRF-13")
    source_reference: str = Field(default="")
    use_field_direction_only: bool = Field(default=True)

    _SUPPORTED_MODELS: ClassVar[set[str]] = {
        "IGRF-13",
        "IGRF13",
        "IGRF_13",
    }

    @field_validator("model")
    @classmethod
    def _validate_model(cls, value: str) -> str:
        """Validate the magnetic-field model name."""
        model_name = str(value).strip()
        if not model_name:
            raise ValueError("data.magnetic_field.model must be non-empty.")
        if model_name.upper() not in {item.upper() for item in cls._SUPPORTED_MODELS}:
            raise ValueError(
                "Unsupported magnetic-field model "
                f"{model_name!r}; expected IGRF-13 for this reproduction."
            )
        return model_name

    @field_validator("use_field_direction_only")
    @classmethod
    def _validate_direction_only(cls, value: bool) -> bool:
        """Validate paper-required use of magnetic-field direction only."""
        if not value:
            raise ValueError(
                "data.magnetic_field.use_field_direction_only must be true; "
                "the BP plane uses magnetic-field direction, not magnitude."
            )
        return value

class _DistanceGridConfig(_StrictBaseModel):
    """Distance-grid settings."""

    min_km: float = Field(default=constants.DEFAULT_BP_MIN_DISTANCE_KM)
    max_km: float = Field(default=constants.DEFAULT_BP_MAX_DISTANCE_KM)
    step_km: float = Field(default=constants.DEFAULT_BP_STEP_KM)

    @field_validator("min_km", "max_km", "step_km")
    @classmethod
    def _validate_finite_positive_components(cls, value: float) -> float:
        """Validate finite grid values before cross-field checks."""
        value = float(value)
        if value != value:
            raise ValueError("Distance-grid values must be finite, not NaN.")
        return value

    @model_validator(mode="after")
    def _validate_grid(self) -> "_DistanceGridConfig":
        """Validate distance-grid ordering and integer stepping."""
        if self.min_km <= 0.0:
            raise ValueError("distance_grid.min_km must be > 0.")
        if self.max_km <= self.min_km:
            raise ValueError("distance_grid.max_km must be > distance_grid.min_km.")
        if self.step_km <= 0.0:
            raise ValueError("distance_grid.step_km must be > 0.")

        span = self.max_km - self.min_km
        ratio = span / self.step_km
        if abs(ratio - round(ratio)) > 1.0e-9:
            raise ValueError(
                "Distance grid must be compatible with integer stepping: "
                "(max_km - min_km) / step_km must be an integer."
            )
        return self

            raise ValueError(

        span = self.max_km - self.min_km
        ratio = span / self.step_km
        if abs(ratio - round(ratio)) > 1.0e-9:
            raise ValueError(
                "Magnetic-field candidate grid must be compatible with integer "
                "stepping: (max_km - min_km) / step_km must be an integer."

        return self

class _BackPropagationConfig(_StrictBaseModel):
    """Back-propagation method configuration."""

    representation: str = Field(default="coordinate")
    propagation_method_default: str = Field(default=constants.DEFAULT_PROPAGATION_METHOD)
    alternative_method_available: str = Field(
        default=constants.DEFAULT_ALTERNATIVE_PROPAGATION_METHOD
    )
    distance_grid: _DistanceGridConfig = Field(default_factory=_DistanceGridConfig)
    magnetic_field_candidate_grid: _MagneticFieldCandidateGridConfig = Field(
        default_factory=_MagneticFieldCandidateGridConfig
    )
    stationary_transmitter_correction: _StationaryTransmitterCorrectionConfig = Field(
        default_factory=_StationaryTransmitterCorrectionConfig
    )
    wavefront_curvature_correction: _WavefrontCurvatureCorrectionConfig = Field(
        default_factory=_WavefrontCurvatureCorrectionConfig
    )
    v_curve: _VCurveConfig = Field(default_factory=_VCurveConfig)
    minimum_detection: _MinimumDetectionConfig = Field(
        default_factory=_MinimumDetectionConfig
    )
    quality_control: _QualityControlConfig = Field(default_factory=_QualityControlConfig)
    geolocation: dict[str, Any] = Field(default_factory=dict)

    @model_validator(mode="after")
    def _validate_cross_field_consistency(self) -> "_BackPropagationConfig":
        """Validate BP and magnetic-field grid consistency."""
        if abs(self.magnetic_field_candidate_grid.step_km - self.distance_grid.step_km) > 1.0e-9:
            raise ValueError(
                "The paper uses the same 100 km step for BP and magnetic-field "
                "candidate loops; magnetic_field_candidate_grid.step_km must "
                "equal distance_grid.step_km."
            )

        if self.v_curve.smoothing.window_km < self.distance_grid.step_km:
            raise ValueError(
                "back_propagation.v_curve.smoothing.window_km must be at least "
                "distance_grid.step_km."
            )

        if (
            abs(self.minimum_detection.sample_spacing_km - self.distance_grid.step_km)
            > 1.0e-9
        ):
            raise ValueError(
                "minimum_detection.sample_spacing_km must equal "
                "distance_grid.step_km for consistent L1/L2 span checks."
            )

        if self.representation.strip().lower() != "coordinate":
            raise ValueError(
                "back_propagation.representation must be 'coordinate' to match "
                "the paper's BP coordinate representation."
            )

        return self

            raise ValueError(

                "because the paper uses 3° x 3° bins."
            )
        return self

class AppConfig(BaseModel):
    """Flat application configuration used by processing modules.

    Attributes:
        cosmic_data_dir: Directory containing COSMIC-2 high-rate phase/SNR data.
        receiver_orbit_dir: Directory containing COSMIC-2 receiver orbit data.
        gnss_orbit_dir: Directory containing GNSS orbit/clock products.
        output_dir: Directory where outputs should be written.
        igrf_model: Magnetic-field model name, expected to be ``"IGRF-13"``.
        sigma_phi_threshold_rad: Pre-BP phase-scintillation threshold in rad.
        tangent_height_min_km: Minimum tangent-point height in km.
        bp_min_distance_km: Minimum BP distance-grid value in km.
        bp_max_distance_km: Maximum BP distance-grid value in km.
        bp_step_km: BP distance-grid step in km.
        mf_step_km: Magnetic-field candidate distance step in km.
        q_threshold: Minimum-depth metric threshold.
        cos_alpha_threshold: Scan-angle cosine threshold.
        smoothing_window_km: V(L) smoothing window in km.
        smoothing_poly_order: Sliding polynomial smoothing order.
        map_bin_deg: Monthly map latitude/longitude bin size in degrees.
        random_seed: Reproducibility seed for synthetic experiments.
    """

    model_config = ConfigDict(
        extra="forbid",
        arbitrary_types_allowed=True,
        validate_assignment=True,
    )

    cosmic_data_dir: Path = Field(default=Path("data/cosmic2/hr"))
    receiver_orbit_dir: Path = Field(default=Path("data/cosmic2/orbits"))
    gnss_orbit_dir: Path = Field(default=Path("data/gnss"))
    output_dir: Path = Field(default=Path("outputs"))

    igrf_model: str = Field(default="IGRF-13")

    sigma_phi_threshold_rad: float = Field(
        default=constants.DEFAULT_SIGMA_PHI_THRESHOLD_RAD
    )
    tangent_height_min_km: float = Field(default=constants.DEFAULT_TANGENT_HEIGHT_MIN_KM)

    bp_min_distance_km: float = Field(default=constants.DEFAULT_BP_MIN_DISTANCE_KM)
    bp_max_distance_km: float = Field(default=constants.DEFAULT_BP_MAX_DISTANCE_KM)
    bp_step_km: float = Field(default=constants.DEFAULT_BP_STEP_KM)
    mf_step_km: float = Field(default=constants.DEFAULT_MF_STEP_KM)

    q_threshold: float = Field(default=constants.DEFAULT_Q_THRESHOLD)
    cos_alpha_threshold: float = Field(default=constants.DEFAULT_COS_ALPHA_THRESHOLD)

    smoothing_window_km: float = Field(default=constants.DEFAULT_SMOOTHING_WINDOW_KM)
    smoothing_poly_order: int = Field(default=constants.DEFAULT_SMOOTHING_POLY_ORDER)

    map_bin_deg: float = Field(default=constants.DEFAULT_MAP_BIN_DEG)
    random_seed: int = Field(default=constants.DEFAULT_RANDOM_SEED)

    @field_validator(
        "cosmic_data_dir",
        "receiver_orbit_dir",
        "gnss_orbit_dir",
        "output_dir",
        mode="before",
    )
    @classmethod
    def _validate_path(cls, value: Any) -> Path:
        """Validate path fields without requiring filesystem existence."""
        if value is None:
            raise ValueError("Configuration path fields must not be null.")
        path = Path(value).expanduser()
        if str(path).strip() == "":
            raise ValueError("Configuration path fields must be non-empty.")
        return path

    @field_validator("igrf_model")
    @classmethod
    def _validate_igrf_model(cls, value: str) -> str:
        """Validate the IGRF model value."""
        model_name = str(value).strip()
        if not model_name:
            raise ValueError("igrf_model must be non-empty.")
        if model_name.upper() not in {"IGRF-13", "IGRF13", "IGRF_13"}:
            raise ValueError(

    @model_validator(mode="after")
    def _validate_cross_field_consistency(self) -> "AppConfig":
        """Validate flat configuration consistency."""
        if self.bp_min_distance_km <= 0.0:
            raise ValueError("bp_min_distance_km must be > 0.")
        if self.bp_max_distance_km <= self.bp_min_distance_km:
            raise ValueError("bp_max_distance_km must be greater than bp_min_distance_km.")
        if self.bp_step_km <= 0.0:
            raise ValueError("bp_step_km must be > 0.")
        if self.mf_step_km <= 0.0:
            raise ValueError("mf_step_km must be > 0.")

        span = self.bp_max_distance_km - self.bp_min_distance_km
        ratio = span / self.bp_step_km
        if abs(ratio - round(ratio)) > 1.0e-9:
            raise ValueError(
                "(bp_max_distance_km - bp_min_distance_km) / bp_step_km "
                "must be an integer."
            )

        if abs(self.mf_step_km - self.bp_step_km) > 1.0e-9:
            raise ValueError(
                "mf_step_km must equal bp_step_km; the paper uses 100 km for "
                "both nested BP loops."
            )

        if self.smoothing_window_km < self.bp_step_km:
            raise ValueError("smoothing_window_km must be >= bp_step_km.")

        return self

    @classmethod
    def from_yaml(cls, path: Path) -> "AppConfig":
        """Load and validate application configuration from a YAML file.

        Args:
            path: Path to a YAML configuration file. The loader is path-driven
                and does not assume a fixed filename.

        Returns:
            Validated ``AppConfig`` instance.

        Raises:
            ConfigLoadError: If the file is missing, YAML syntax is invalid,
                the top-level YAML object is malformed, or validation fails.
        """
        config_path = Path(path).expanduser()
        if not config_path.exists():
            raise ConfigLoadError(f"Configuration file does not exist: {config_path}")
        if not config_path.is_file():
            raise ConfigLoadError(f"Configuration path is not a file: {config_path}")

        try:
            with config_path.open("r", encoding="utf-8") as stream:
                loaded_yaml = yaml.safe_load(stream)
        except yaml.YAMLError as exc:
            raise ConfigLoadError(
                f"Invalid YAML syntax in configuration file {config_path}: {exc}"
            ) from exc
        except OSError as exc:
            raise ConfigLoadError(
                f"Could not read configuration file {config_path}: {exc}"
            ) from exc

        if loaded_yaml is None:
            raise ConfigLoadError(f"Configuration file is empty: {config_path}")
        if not isinstance(loaded_yaml, dict):
            raise ConfigLoadError(
                "Configuration top-level YAML object must be a mapping/dictionary."
            )

        try:
            raw_config = _RawConfig.model_validate(loaded_yaml)
            app_config = cls._from_raw_config(raw_config)
        except ValidationError as exc:
            raise ConfigLoadError(
                f"Configuration validation failed for {config_path}:\n{exc}"
            ) from exc
        except ValueError as exc:
            raise ConfigLoadError(
                f"Configuration validation failed for {config_path}: {exc}"
            ) from exc

        return app_config

    @classmethod
    def _from_raw_config(cls, raw_config: _RawConfig) -> "AppConfig":
        """Map the internal nested YAML model to the flat public config."""
        latitude_bin_deg = raw_config.real_data_experiments.monthly_l1_maps.bin_size_deg.latitude
        longitude_bin_deg = raw_config.real_data_experiments.monthly_l1_maps.bin_size_deg.longitude
        if abs(latitude_bin_deg - longitude_bin_deg) > 1.0e-12:
            raise ValueError(
                "Monthly map latitude and longitude bin sizes must be equal."
            )

        return cls(
            cosmic_data_dir=raw_config.paths.cosmic2_hr_data_dir,
            receiver_orbit_dir=raw_config.paths.receiver_orbit_dir,
            gnss_orbit_dir=raw_config.paths.gnss_orbit_clock_dir,
            output_dir=raw_config.paths.output_dir,
            igrf_model=raw_config.data.magnetic_field.model,
            sigma_phi_threshold_rad=(
                raw_config.processing.scintillation_indices.sigma_phi_threshold_rad
            ),
            tangent_height_min_km=(
                raw_config.processing.tangent_point_filter.min_height_km
            ),
            bp_min_distance_km=raw_config.back_propagation.distance_grid.min_km,
            bp_max_distance_km=raw_config.back_propagation.distance_grid.max_km,
            bp_step_km=raw_config.back_propagation.distance_grid.step_km,
            mf_step_km=(
                raw_config.back_propagation.magnetic_field_candidate_grid.step_km
            ),
            q_threshold=raw_config.back_propagation.quality_control.q_threshold,
            cos_alpha_threshold=(
                raw_config.back_propagation.quality_control.cos_alpha_threshold
            ),
            smoothing_window_km=(
                raw_config.back_propagation.v_curve.smoothing.window_km
            ),
            smoothing_poly_order=(
                raw_config.back_propagation.v_curve.smoothing.polynomial_order
            ),
            map_bin_deg=latitude_bin_deg,
            random_seed=raw_config.reproducibility.random_seed,
        )
